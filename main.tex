\section{What Is Autoresearch?}

\begin{frame}{}
  \centering
  \Large What Is Autoresearch?
\end{frame}

\begin{frame}{What Is Autoresearch?}
  \begin{itemize}
    \item Using AI agents to automate the \textbf{research process itself}
    \item Distinct from AI applied to a domain problem (medical diagnosis, protein folding, image generation)
    \item The agent generates ideas, designs and runs experiments, writes papers
    \item This talk: survey the current landscape --- systems, techniques, and limitations
  \end{itemize}
\end{frame}

\section{The Research Pipeline}

\begin{frame}{}
  \centering
  \Large The Research Pipeline
\end{frame}

\begin{frame}{The Research Lifecycle}
  \begin{enumerate}
    \item Idea Generation \& Literature Review
    \item Research Planning \& Experiment Design
    \item Data Preparation \& Experiment Execution
    \item Results Analysis
    \item Report Writing \& Synthesis
    \item Paper Review
  \end{enumerate}
\end{frame}

\begin{frame}{Stage 1: Idea Generation}
  \begin{itemize}
    \item LLM brainstorms ideas from a research prompt or template
    \item \textbf{Novelty filtering}: system queries Semantic Scholar API and filters out ideas too similar to existing work
    \item \textbf{Scoring rubric}: each idea rated 1--10 on Interestingness, Feasibility, and Novelty
    \item Multiple rounds of \textbf{chain-of-thought + self-reflection} refine and diversify the idea set
    \item A running archive prevents re-proposing already-explored ideas
  \end{itemize}
  \vfill
  \footnotesize\citep{lu2024aiscientist}
\end{frame}

\begin{frame}{Stage 2: Experimentation}
  \begin{itemize}
    \item \textbf{Aider} (LLM coding agent) implements plan-directed code changes; runs experiments automatically
    \item \textbf{v1 (AI Scientist)}: linear loop --- implement $\to$ run $\to$ analyze $\to$ re-plan, up to 5 iterations
    \item \textbf{v2 (AI Scientist-v2)}: tree search --- parallel branches, best-first selection guided by an LLM evaluator; vision-language model (VLM) checks generated figures
    \item Model comparison on v1 (best $\to$ worst): Claude Sonnet 3.5, GPT-4o, DeepSeek Coder, Llama-3.1 405B; full run costs \textasciitilde\$10
  \end{itemize}
  \vfill
  \footnotesize\citep{lu2024aiscientist, yamada2025aiscientistv2}
\end{frame}

\begin{frame}{Stage 3: Paper Writing \& Review}
  \begin{itemize}
    \item Paper written \textbf{section-by-section} by the LLM; LaTeX errors fed back for automated fixing
    \item \textbf{Semantic Scholar API} retrieves references for the related work section
    \item \textbf{Automated peer review}: GPT-4o reviewer with 5-round self-reflection and 5 ensembled reviews
    \item Reviewer accuracy: \textbf{65\% balanced accuracy} vs.\ 66\% for human reviewers (near-human) \citep{lu2024aiscientist}
    \item Cost: \textbf{\$0.25--\$0.50 per review}; \textbf{under \$15 per full paper}
  \end{itemize}
\end{frame}

\section{Case Studies}

\begin{frame}{}
  \centering
  \Large Case Studies
\end{frame}

\begin{frame}{Case Study 1: The AI Scientist}
  \begin{itemize}
    \item End-to-end machine learning autoresearch: idea generation $\to$ experiments $\to$ paper \citep{lu2024aiscientist}
    \item \textbf{Existence proof}: first AI-generated paper accepted at peer review\\
          (ICLR 2025 ICBINB workshop; 1 of 3 submissions accepted) \citep{yamada2025aiscientistv2}
    \item Cost: \textbf{\$15 per paper end-to-end}
    \item Human role: configure the system once, then step away
  \end{itemize}
\end{frame}

\begin{frame}{Case Study 2: Coscientist}
  \begin{itemize}
    \item GPT-4 agent equipped with four tools: web search, documentation reader, code executor, robotic lab controller \citep{boiko2023coscientist}
    \item Task: autonomously design \textit{and physically execute} a palladium-catalyzed cross-coupling reaction
    \item \textbf{Key milestone}: crossed the digital--physical boundary --- autonomous wet-lab experiment execution, not just writing
    \item Human role: define the research goal, then step away
  \end{itemize}
\end{frame}

\begin{frame}{Case Study 3: The Virtual Lab}
  \begin{itemize}
    \item Goal: design SARS-CoV-2 nanobodies targeting JN.1/KP.3 variants \citep{swanson2024virtuallab}
    \item \textbf{Structure}: human PI writes meeting agendas; LLM PI agent leads specialist agents (immunologist, ML specialist, computational biologist, critic)
    \item Human involvement: $\approx$\textbf{1\% of interactions} --- but human controls direction at every meeting boundary
    \item Result: 92 nanobodies designed computationally; 2 showed improved binding \textit{and} maintained ancestral-spike binding; published in \textit{Nature} 2025
    \item \textbf{Key contrast}: fully autonomous systems hand off execution entirely; Virtual Lab keeps the human as PI --- agenda-setting and validation stay human
  \end{itemize}
\end{frame}

\section{Limitations and Risks}

\begin{frame}{}
  \centering
  \Large Limitations and Risks
\end{frame}

\begin{frame}{Technical Failures}
  \begin{itemize}
    \item \textbf{Literature review} is the most consistently failing stage across current systems \citep{schmidgall2025agenticscience}
    \item \textbf{Hallucinated citations}: AI Scientist-v2 introduced citation inaccuracies even in its accepted paper \citep{yamada2025aiscientistv2}
    \item \textbf{Positive-spin bias}: tendency to emphasize positive results and underreport negative findings \citep{yamada2025aiscientistv2}
    \item \textbf{Brittleness on novel tasks}: agents perform well on well-defined problems, struggle with genuinely new hypotheses \citep{feng2026visionary}
    \item \textbf{Workshop-level, not conference-level}: 1 of 3 AI Scientist-v2 papers accepted --- at a workshop (60--80\% acceptance rate), not a main-track venue \citep{yamada2025aiscientistv2}
  \end{itemize}
\end{frame}

\begin{frame}{Systemic Risks}
  \begin{itemize}
    \item \textbf{Verification asymmetry}: checking AI-generated work requires more focused effort than producing it \citep{feng2026visionary}
    \item \textbf{Literature flooding}: agents can generate plausible-looking papers at scale --- polished mediocrity saturates venues \citep{feng2026visionary}
    \item \textbf{Physical safety}: agents integrated with robotic labs must have well-defined blast-radius constraints; misinterpreting physical context can cause real harm \citep{schmidgall2025agenticscience}
    \item \textbf{Weak disclosure norms}: systems could inflate CVs or obscure AI authorship; the AI Scientist-v2 authors withdrew accepted papers to avoid setting a precedent \citep{yamada2025aiscientistv2}
  \end{itemize}
\end{frame}

\section{Conclusions}

\begin{frame}{}
  \centering
  \Large Conclusions
\end{frame}

\begin{frame}{Conclusions}
  \begin{itemize}
    \item Autoresearch has crossed the \textbf{existence-proof threshold}: AI-generated work
          now passes peer review, and autonomous agents can execute real wet-lab experiments
    \item \textbf{Cost and speed are not the bottleneck} --- \$15/paper end-to-end;
          the limits are quality, novelty, and trust, not compute
    \item The \textbf{hybrid model may be more realistic near-term} than fully autonomous:
          Virtual Lab's $\approx$1\% human contribution produced two Nature-validated nanobodies;
          full autonomy produced workshop papers the authors chose to withdraw
    \item \textbf{Systemic risks are underspecified}: verification asymmetry, literature
          flooding, and disclosure norms are unsolved --- the technology is ahead of the norms
    \item The research lifecycle is a \textbf{pipeline, not a single task}; current systems
          fail most predictably on literature review and on generating genuinely novel hypotheses
  \end{itemize}
\end{frame}
